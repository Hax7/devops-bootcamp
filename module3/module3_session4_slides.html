boto3 — AWS SDK

import boto3
s3 = boto3.client("s3")
print([b["Name"] for b in s3.list_buckets().get("Buckets", [])])

paramiko — SSH

import paramiko
ssh = paramiko.SSHClient(); ssh.set_missing_host_key_policy(paramiko.AutoAddPolicy())
ssh.connect("1.2.3.4", username="ec2-user", key_filename="~/.ssh/key.pem")
print(ssh.exec_command("hostname")[1].read().decode()); ssh.close()

docker SDK — Containers

import docker
cli = docker.from_env()
print(cli.containers.run("nginx:latest", detach=True).short_id)

kubernetes — K8s API

from kubernetes import client, config
config.load_kube_config(); v1 = client.CoreV1Api()
v1.delete_namespaced_pod(name="example-pod", namespace="default")

pyyaml — YAML

import yaml
data = yaml.safe_load(open("deployment.yaml"))
data["spec"]["replicas"] = 5
yaml.safe_dump(data, open("deployment.yaml","w"))

requests — HTTP APIs

import requests, os
requests.post(os.environ["SLACK_WEBHOOK"], json={"text":"Deploy OK"}, timeout=8).raise_for_status()

fabric — SSH tasks

from fabric import Connection
c = Connection("ubuntu@myserver.com")
c.run("cd /srv/app && git pull && docker compose up -d")

pytest — tests

import requests
def test_homepage():
    assert requests.get("http://localhost:8000").status_code == 200

ansible-runner — run playbooks

import ansible_runner
r = ansible_runner.run(private_data_dir="/ansible", playbook="deploy.yml")
print(r.status, r.rc)

sys — CLI plumbing

import sys
if len(sys.argv) != 2: print("Usage: deploy.py "); sys.exit(1)
